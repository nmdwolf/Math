\chapter{Complex bundles}

\section{Almost complex structure}

	\newdef{Almost complex structure}{
		Let $M$ be a real manifold. An almost complex strcuture on $M$ is a smooth $(1, 1)$-tensor field $J:TM\rightarrow TM$ such that $J|_p:T_pM\rightarrow T_pM$ satisifes $J|_p^2 = -1$ for all $p\in M$. An \textbf{almost complex manifold} is a real manifold equipped with an almost complex structure.
	}
	
	\begin{property}
		An almost complex manifold is even-dimensional and orientable.
	\end{property}
	
	\begin{property}
		A manifold $M$ admits an almost complex structure if and only if the structure group of the tangent bundle $TM$ can be reduced from $GL(\mathbb{R}^{2n})$ to $GL(\mathbb{C}^n)$.
	\end{property}

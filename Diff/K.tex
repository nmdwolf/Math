\chapter{K-theory}

\section{Basic definitions}

	\newdef{K-theory}{
		Let $X$ be a compact Hausdorff space and let Vect$(X)$ be the set of isomorphism classes of finite-dimensional vector bundles over $X$. Because this set is well-behaved with respect to Whitney sums, the structure $(\text{Vect}(X), \oplus)$ forms an Abelian monoid. The Grothendieck completion\footnote{See definition \ref{group:grothendieck_completion}.} of $(\text{Vect}(X), \oplus)$ is called the $K$-theory of $X$.
	}
	
	\begin{notation}
		The $K$-theory of a space $X$ is denoted by $K^0(X)$.
	\end{notation}

\chapter{Derivations: Quantum mechanics}
\label{derivations:QM}

	\newformula{TDSE and TISE}{
    	Starting from the one-dimensional TDSE in position space with a time-independent potential
        \begin{equation}
			i\hbar\pderiv{}{t}\Psi(x,t) = \left(-\stylefrac{\hbar^2}{2m}\partial_x^{\ 2} + V(x)\right)\Psi(x, t)
		\end{equation}
        we try separation of variables and assert a solution of the form $\Psi(x, t) = \psi(x)T(t)$.
        Inserting this in previous equation gives:
        \begin{equation}
			i\hbar \psi(x)T'(t) = -\stylefrac{\hbar^2}{2m}T(t)\psi''(x) + V(x)\psi(x)T(t)
		\end{equation}
        Dividing both sides by $X(x)T(t)$ and rearranging the terms gives:
        \begin{equation}
			i\hbar\stylefrac{T'(t)}{T(t)} = \left(-\stylefrac{\hbar^2}{2m} + V(x)\right)\stylefrac{\psi''(x)}{\psi(x)}
		\end{equation}
        As the left side only depends on $t$ and the right side only on $x$, we conclude that they both equal a constant $E$. This leads us to the following system of differential equations:
        \begin{equation}
            \begin{cases}
                i\hbar T'(t) - ET(t) = 0\\
                \left(-\stylefrac{\hbar^2}{2m} + V(x)\right)\psi''(x) = E\psi(x)
            \end{cases}
        \end{equation}
        The first equation immediately gives a solution for $T(t)$:
        \begin{equation}
        	\label{derivations:qm:exponential}
			\boxed{T(t) = Ce^{-\frac{i}{\hbar}Et}}
		\end{equation}
        Rearranging the second solution in the system gives the TISE:
        \begin{equation}
        	\label{derivations:qm:TISE}
			\boxed{\psi''(x) = -\stylefrac{2m}{\hbar^2}\left(E - V(x)\right)\psi(x)}
		\end{equation}
    }

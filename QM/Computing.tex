\chapter{Entanglement \& Quantum computing}

\section{Bipartite systems}
\subsection{Marginal density operators}

	\newdef{Marginal density operator}{\index{density!operator}
		Let $|\Psi\rangle_{AB}$ be the state of a bipartite system. The marginal density operator $\op\rho_A$ of system A is defined as follows:
		\begin{equation}
			\op\rho_A = \text{Tr}_B|\Psi\rangle_{ABAB}\langle\Psi|
		\end{equation}
	}
	
	\newdef{Purification}{\index{purification}
		Let $\op\rho_A$ be the desnity operator of a system A. A purification of $\op\rho_A$ is a pure state $|\Psi\rangle_{AB}$ of a composite system $AB$ such that:
		\begin{equation}
			\op\rho_A = \text{Tr}_B|\Psi\rangle_{ABAB}\langle\Psi|
		\end{equation} 	
	}
	\begin{property}
		Any two purifications of the same density operator $\hat\rho_A$ are related by a transformation $\mathbbm{1}_A\otimes\op{V}$, where $\op{V}$ is a unitary operator on $\mathcal{H}_B$. 
	\end{property}

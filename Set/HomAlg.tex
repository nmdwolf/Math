\section{Spectral sequences}

	\newdef{Spectral sequence}{\index{spectral sequence}
		Consider a sequence of differential objects $\{(A_i, d_i)\}$. This sequence is called a spectral sequence if it satisfies
		\begin{equation}
			H(A_i, d_i) = E_{i+1}
		\end{equation}
		for every $i$. A morphism of spectral sequence is a collection of morphism $\{\varphi_i\}$ satisfying:
		\begin{itemize}
			\item $\varphi_i\circ d_i = d_i'\circ\varphi_i$
			\item $\varphi_{i+1} = H(\varphi_i)$
		\end{itemize}
	}
